Quick Integration Guide:
UpdateImu(rad): Trigger this inside your IMU callback or a dedicated 25 Hz timer. Use a Stopwatch-based timer for high precision.
UpdateGps(...): Trigger this as soon as your NMEA parser (or AgOpenGPS equivalent) extracts a new coordinate and speed.
AntennaHeight: This variable is crucial. If your antenna is 3.2m above ground, a 5-degree roll error shift is roughly 28cm.
Output: Use PosX and PosY for your map drawing and line-following logic; they represent the compensated ground-center of the tractor.

using System;
using System.Diagnostics;
using System.Linq;

namespace AgOpenGPS_Fusion
{
    public class AgFusionEngine
    {
        // --- Configuration (Adjust based on your tractor) ---
        private const double AntennaHeight = 3.2;      // Height from ground to antenna (meters)
        private const double PosGain = 0.6;           // 0.6 = Trust GPS 60%, Dead Reckoning 40%
        private const double HeadingGain = 0.15;      // How fast GPS COG corrects IMU heading
        private const double MinSpeedForHeading = 0.8; // m/s required to trust GPS COG
        private const int SpeedWindowSize = 5;         // 5 samples = 0.5s smoothing at 10Hz

        // --- State Variables ---
        public double PosX { get; private set; }      // Ground-center Easting (m)
        public double PosY { get; private set; }      // Ground-center Northing (m)
        public double Heading { get; private set; }   // True North Radians
        private double _smoothedSpeed;
        
        private double _lastGpsX, _lastGpsY;
        private bool _isInitialized = false;

        // --- High-Resolution Timing & Smoothing ---
        private long _lastTicks;
        private static readonly double TicksToSec = 1.0 / Stopwatch.Frequency;
        private readonly double[] _speedBuffer = new double[SpeedWindowSize];
        private int _bufferIndex = 0;

        /// <summary>
        /// Update IMU: Called at 25 Hz.
        /// Performs high-precision dead reckoning extrapolation between GPS pulses.
        /// </summary>
        /// <param name="imuHeadingRad">Heading from IMU in Radians</param>
        public void UpdateImu(double imuHeadingRad)
        {
            long currentTicks = Stopwatch.GetTimestamp();
            
            // Initialization for timing
            if (_lastTicks == 0) 
            { 
                _lastTicks = currentTicks; 
                Heading = imuHeadingRad;
                return; 
            }

            // Calculate exact elapsed time using high-res performance counter
            double dt = (currentTicks - _lastTicks) * TicksToSec;
            _lastTicks = currentTicks;

            // Sync internal heading state
            Heading = imuHeadingRad;

            // Predict current position forward based on last known smoothed speed
            double distance = _smoothedSpeed * dt;
            PosX += Math.Sin(Heading) * distance;
            PosY += Math.Cos(Heading) * distance;
        }

        /// <summary>
        /// Update GPS: Called at 10 Hz.
        /// Performs Terrain Compensation (Roll/Pitch) and corrects Dead Reckoning drift.
        /// </summary>
        /// <param name="rawGpsX">Raw GPS Easting (m)</param>
        /// <param name="rawGpsY">Raw GPS Northing (m)</param>
        /// <param name="rawGpsSpeedMps">GPS Ground Speed (m/s)</param>
        /// <param name="rollRad">Roll from IMU (Positive = Tilt Right)</param>
        /// <param name="pitchRad">Pitch from IMU (Positive = Nose Down)</param>
        public void UpdateGps(double rawGpsX, double rawGpsY, double rawGpsSpeedMps, double rollRad, double pitchRad)
        {
            // 1. Terrain Compensation (Project Antenna to Ground Center)
            // Roll corrects side-to-side (90 deg to heading)
            // Pitch corrects forward-back (Parallel to heading)
            double rollDist = Math.Sin(rollRad) * AntennaHeight;
            double pitchDist = Math.Sin(pitchRad) * AntennaHeight;

            double correctedGpsX = rawGpsX - (Math.Cos(Heading) * rollDist) - (Math.Sin(Heading) * pitchDist);
            double correctedGpsY = rawGpsY + (Math.Sin(Heading) * rollDist) - (Math.Cos(Heading) * pitchDist);

            // 2. Speed Smoothing (Moving Average)
            _speedBuffer[_bufferIndex] = rawGpsSpeedMps;
            _bufferIndex = (_bufferIndex + 1) % _speedBuffer.Length;
            _smoothedSpeed = _speedBuffer.Average();

            if (!_isInitialized)
            {
                PosX = correctedGpsX; 
                PosY = correctedGpsY;
                _lastGpsX = correctedGpsX; 
                _lastGpsY = correctedGpsY;
                _isInitialized = true;
                return;
            }

            // 3. Fusion: Pull the IMU prediction back toward the absolute ground-center GPS fix
            PosX = (1.0 - PosGain) * PosX + (PosGain * correctedGpsX);
            PosY = (1.0 - PosGain) * PosY + (PosGain * correctedGpsY);

            // 4. Heading Correction (Using Course Over Ground)
            if (_smoothedSpeed > MinSpeedForHeading)
            {
                // Calculate bearing from last corrected GPS position
                double gpsCourse = Math.Atan2(correctedGpsX - _lastGpsX, correctedGpsY - _lastGpsY);
                
                // Calculate shortest-path angle difference (wraps 0/2PI)
                double diff = (gpsCourse - Heading + Math.PI) % (2 * Math.PI) - Math.PI;
                if (diff < -Math.PI) diff += 2 * Math.PI;
                
                // Nudge IMU heading toward GPS truth to eliminate gyro drift
                Heading += diff * HeadingGain;
            }

            // Save state for next GPS update
            _lastGpsX = correctedGpsX;
            _lastGpsY = correctedGpsY;

            // Zero Velocity Update (ZUPT) - Stop ghost movement when stationary
            if (_smoothedSpeed < 0.1) _smoothedSpeed = 0;
        }

        /// <summary>
        /// Resets the engine (useful if GPS signal is lost for a long time)
        /// </summary>
        public void Reset()
        {
            _isInitialized = false;
            _lastTicks = 0;
            Array.Clear(_speedBuffer, 0, _speedBuffer.Length);
        }
    }
}
